function keyPressed() {
    if(keyCode == 32) {
        allowedToStop = true;
    }
    if(keyCode == 48) {
        allowedToStart = true;
    }
}

function drawThing() {
    translate(-width, 0, -height);

    for(let i = 0; i < 10; i++) {

        for(let j = 0; j < 10; j++) {
            push();

            translate(100*i, 0, 100*j);

            push();
            rotateY(frameCount/100);

            // fill(palette[(i+j + ~~(frameCount/10))%5]);
            fill(palette[(i+j)%5]);
            plane(100, 100);
            pop();

            rotateY(radians(90));

            push();
            rotateY(frameCount/100);
            // fill(palette[(1 + i+j + ~~(frameCount/10))%5]);
            fill(palette[(1 + i+j)%5]);
            plane(101, 101);
            pop();

            pop();
        }


    }
}

function keyPressed() {
    if(keyCode == 32) {
        palette = randomPalette1000();
    }
}
